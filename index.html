async function analyzeAndLog(blob) { recordBtn.innerText = "WAIT"; statusText.innerText = "Processing sound..."; // Safety check: ensure blob has data if (blob.size < 1000) { statusText.innerText = "Error: Recording too short"; return; } const formData = new FormData(); formData.append('audio', blob, 'recording.wav'); try { // Use a cache-buster (?t=) to prevent the proxy from serving a stale 'Error' page const target = 'https://birdnet.cornell.edu/api/v1/analyze'; const proxy = `https://api.allorigins.win/raw?url=${encodeURIComponent(target)}&t=${Date.now()}`; const response = await fetch(proxy, { method: 'POST', body: formData }); if (!response.ok) throw new Error(`HTTP ${response.status}`); const rawText = await response.text(); // Isolate the JSON object const start = rawText.indexOf('{'); const end = rawText.lastIndexOf('}') + 1; if (start === -1) throw new Error("Invalid Server Response"); const birdData = JSON.parse(rawText.substring(start, end)); if (birdData.results && birdData.results.length > 0) { const name = birdData.results[0].species_common_name; const conf = Math.round(birdData.results[0].confidence * 100); statusText.innerText = `${name} (${conf}%)`; // Log to Google Sheet fetch(SHEET_URL, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: `bird=${encodeURIComponent(name)}&confidence=${conf}&audioUrl=Logged` }); updateUI(name, conf); } else { statusText.innerText = "No bird detected."; } } catch (e) { console.error("Full Error:", e); // Show specific error to help us debug statusText.innerText = "ID Failed: " + e.message; } finally { recordBtn.innerText = "LISTEN"; recordBtn.style.background = "var(--green)"; isRecording = false; } }
