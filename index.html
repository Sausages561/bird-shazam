const PIPEDREAM_URL = 'https://eoqh468oz3nen4s.m.pipedream.net'; const btn = document.getElementById('btn'); const status = document.getElementById('status'); btn.onclick = async () => { try { const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); const recorder = new MediaRecorder(stream); const chunks = []; recorder.ondataavailable = e => chunks.push(e.data); recorder.onstop = async () => { status.innerText = "Transmitting to Bridge..."; const blob = new Blob(chunks, { type: 'audio/wav' }); const reader = new FileReader(); reader.readAsDataURL(blob); reader.onloadend = async () => { const base64Data = reader.result.split(',')[1]; await fetch(PIPEDREAM_URL, { method: 'POST', body: base64Data }); status.innerText = "Processing BirdNET..."; }; }; recorder.start(); btn.style.background = "#ff4444"; setTimeout(() => { recorder.stop(); stream.getTracks().forEach(t => t.stop()); btn.style.background = "#1DB954"; }, 3000); // 3-second sample } catch (e) { status.innerText = "Mic Error: " + e.message; } };
